oracle创建表空间并赋值给用户

创建表空间

create tablespace eprk
logging datafile 'D:\oracle\oradata\orcl\eprk.dbf'#表空间的路径
size 100m autoextend on
next 50m maxsize 20480m extent management local;

创建qh2用户并指定表空间

create user qh2 identified by 123456
default tablespace eprk
temporary tablespace temp profile default;

给qh2用户授予权限

grant connect,resource to qh2;
grant dba to qh2;

远程导入dmp文件

cd /d D:\oraclexe\app\oracle\product\11.2.0\server\bin

imp qh2/123456@192.168.0.1:1521/orcl file=G:\global.dmp full=y
