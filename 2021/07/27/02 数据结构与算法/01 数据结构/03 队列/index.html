队列

队列介绍

  1. 队列是一个有序列表，可以用**数组**或是**链表**来实现。 

  2. 遵循先入先出的原则。即：**先存入队列的数据，要先取出。后存入的要后取出** 

  3. 示意图：(使用数组模拟队列示意图)

思路

  • front变量的指向队列的第一个元素，初始值为0；
  • rear变量指向队列的最后一个元素的后一个位置，因此每个队列都需要空出一个位置用来做约定
  • 当队满的时候:(rear + 1) %maxSize == front;
  • 当队空的时候，rear == front;
  • 判断队中的元素个数:(rear + maxSize - front) % maxSize;

代码

import java.util.Scanner;

/**
 * @author 作者:张帅
 * @version 创建时间：Nov 13, 2020 9:19:13 PM
 * 
 */
public class CircleArrayQueue {

    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        System.out.print("输入队列的长度：");
        int maxSize = input.nextInt();

        CircleArray circleArray = new CircleArray(maxSize + 1);
        boolean flag = true;
        int tem;
        while (flag) {
            System.out.println("输入要操作的编号：");
            System.out.println("1、打印队列");
            System.out.println("2、数据添加");
            System.out.println("3、数据删除");
            System.out.println("4、查看第一个数字");
            System.out.println("5、退出程序");
            tem = input.nextInt();
            switch (tem) {
            case 1:
                circleArray.printQueue();
                break;
            case 2:
                System.out.println("输入要添加的数字");
                int n = input.nextInt();
                circleArray.addQueue(n);
                break;
            case 3:
                circleArray.putQueue();
                break;
            case 4:
                int peek = circleArray.peek();
                System.out.println(peek);
                break;
            case 5:
                flag = false;
                break;
            }
        }
        System.out.println("程序退出成功");
    }

}

class CircleArray {

    private int maxSize; // 表示数组的最大容量
    private int front; // 变量的含义做一个调整，初始值0，指向当前队列
    private int rear; // 指向队列最后一个元素的下一个

    private int[] arr;

    public CircleArray(int maxSize) {
        this.maxSize = maxSize;
        arr = new int[maxSize];
    }

    // 判断队列是否为空
    public boolean isEmpty() {
        return front == rear;
    }

    // 判断是否满
    public boolean isFull() {
        return (rear + 1) % maxSize == front;
    }

    // 添加数据到队列
    public void addQueue(int n) {
        // 判断队列是否为满
        if (isFull()) {
            System.out.println("队列满，不能加入数据~");
            return;
        }
        // 直接加入数据
        arr[rear] = n;
        // rear数据后移
        rear = (rear + 1) % maxSize;
    }

    // 出队列
    public int putQueue() {
        // 判断队列是否为空
        if (isEmpty()) {
            throw new RuntimeException("队列空");
        }
        int value = arr[front];
        front = (front + 1) % maxSize;
        return value;
    }

    // 输出所有队列中数据
    public void printQueue() {
        // 判断是否为空
        if (isEmpty()) {
            System.out.println("队列空");
            return;
        }
        for (int i = front; i < front + size(); i++) {
            System.out.printf("arr[%d] = %d\n", i % maxSize, arr[i % maxSize]);
        }
    }

    // 队列中 的有效数据
    public int size() {
        return (rear + maxSize - front) % maxSize;
    }

    // 查看队列第一个元素
    public int peek() {
        if (isEmpty()) {
            throw new RuntimeException("队列空");
        }
        return arr[front];
    }
}
