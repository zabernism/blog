import lombok.extern.slf4j.Slf4j;
import org.joda.time.DateTime;
import org.joda.time.format.DateTimeFormat;
import org.joda.time.format.DateTimeFormatter;

import java.text.DateFormat;
import java.text.ParseException;
import java.text.SimpleDateFormat;
import java.time.*;
import java.time.temporal.ChronoUnit;
import java.time.temporal.TemporalAdjusters;
import java.util.Date;
import java.util.Locale;

/**
 * 日期工具类
 */
@Slf4j
public class DateUtil {

    public final static String YMDHMS_SDF = "yyyy-MM-dd HH:mm:ss";
    public final static String YMD_SDF = "yyyy-MM-dd";
    public final static String HMS_SDF = "HH:mm:ss";
    public final static String YMD2_SDF = "yyyyMMdd";
    public final static String START_TIME = " 00:00:00";
    public final static String END_TIME = " 23:59:59";
    public final static String YMD = "yyyyMMdd";
    public final static String YMD_CHINESE = "yyyy年MM月dd日";

    /**
     * 获取当前日期时间 如：2020-10-11 12:21:34
     *
     * @return
     */
    public static String getCurrentDateTimeStamp() {
        return new DateTime().toString(YMDHMS_SDF);
    }

    /**
     * 获取当前日期 如：2020-10-11
     *
     * @return
     */
    public static String getCurrentDateStamp() {
        return new DateTime().toString(YMD_SDF);
    }

    /**
     * 获取当前时间 如：12:21:34
     *
     * @return
     */
    public static String getCurrentTimeStamp() {
        return new DateTime().toString(HMS_SDF);
    }

    /**
     * 获取当前日期 如：20201015
     *
     * @return
     */
    public static String getCurrentDateYMD() {
        return new DateTime().toString(YMD2_SDF);
    }


    /**
     * 将date时间格式转换为yyyy-MM-dd
     *
     * @param dat
     * @return
     */
    public static String getDateYMD(Date dat) {
        return new DateTime(dat).toString(YMD_SDF);
    }

    /**
     * 将date时间格式转换为yyyy-MM-dd
     *
     * @param dat
     * @return
     */
    public static String getStringYMD(Date dat) {
        return getString(dat, YMD);
    }

    public static String getStringYMDSDF(Date dat) {
        return getString(dat, YMD_SDF);
    }

    /**
     * 将date时间格式转换为yyyy-MM-dd HH:mm:ss
     *
     * @param dat
     * @return
     */
    public static String getDateYMDHMS(Date dat) {
        return getString(dat, YMDHMS_SDF);
    }

    public static String getString(Date date, String pattern) {
        return new DateTime(date).toString(pattern);
    }

    public static Date getStrToDateYMDSDF(String strDate) {
        return getStringToDate(strDate, YMD_SDF);
    }

    public static Date getStrToDate(String strDate) {
        return getStringToDate(strDate, YMDHMS_SDF);
    }

    private static Date getStringToDate(String date, String pattern) {
        DateTimeFormatter dateTimeFormatter = DateTimeFormat.forPattern(pattern);
        DateTime dateTime = dateTimeFormatter.parseDateTime(date);
        return dateTime.toDate();
    }

    /**
     * 获取months个月后的时间
     *
     * @param months
     * @return
     */
    public static Date getMonthsDate(Integer months) {
        return new DateTime().plusMonths(months).toDate();
    }

    /**
     * 获取days天后的时间
     *
     * @param days
     * @return
     */
    public static Date getDaysDate(Integer days) {
        return new DateTime().plusDays(days).toDate();
    }

    /**
     * 获取years年后的时间
     *
     * @param dat   传入时间
     * @param years 年
     * @return 时间
     */
    public static Date getDaysDateAfterYears(Date dat, Integer years) {
        return new DateTime(dat).plusYears(years).toDate();
    }

    /**
     * 获取minutes分后的时间
     *
     * @param dat     传入时间
     * @param minutes 分钟
     * @return 时间
     */
    public static Date getMinuteDate(Date dat, Integer minutes) {
        return new DateTime(dat).plusMinutes(minutes).toDate();
    }

    /**
     * 获取day天后的时间
     *
     * @param dat  传入时间
     * @param days 天
     * @return 时间
     */
    public static Date getDayDate(Date dat, Integer days) {
        return new DateTime(dat).plusDays(days).toDate();
    }

    /**
     * 获取指定日期的开始时间
     *
     * @param dat 传入的指定日期
     * @return
     */
    public static String getDayStartTime(String dat) {
        return dat.trim() + START_TIME;
    }

    /**
     * 获取指定日期的结束时间
     *
     * @param dat 传入的指定日期
     * @return
     */
    public static String getDayEndTime(String dat) {
        return dat.trim() + END_TIME;
    }

    public static String getDateString(Date date) {
        SimpleDateFormat formatter = new SimpleDateFormat(YMDHMS_SDF);
        return formatter.format(date);
    }

    /**
     * 2020-06-23T09:46:27.000Z转换成yyyy-MM-dd HH24:mi:ss
     *
     * @param date    时间字符串
     * @param pattern 格式
     * @return 返回结果
     */
    public static String formatData(String date, String pattern) {
        Date temp = null;
        DateFormat afterFormat = null;
        try {
            DateFormat df = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss");
            Date preDate = df.parse(date);
            SimpleDateFormat sdf = new SimpleDateFormat("EEE MMM dd HH:mm:ss Z yyyy", Locale.UK);
            temp = sdf.parse(preDate.toString());
            afterFormat = new SimpleDateFormat(pattern);
        } catch (ParseException e) {
            e.printStackTrace();
        }
        assert afterFormat != null;
        return afterFormat.format(temp);
    }

    public Date genDate(String date, Integer hour, Integer minute) {
        LocalDate localDate = LocalDate.parse(date, java.time.format.DateTimeFormatter.ofPattern(YMD_SDF));
        LocalTime localTime = LocalTime.of(hour, minute);
        return asDate(LocalDateTime.of(localDate, localTime));
    }


    public static Date asDate(LocalDate localDate) {
        return Date.from(localDate.atStartOfDay().atZone(ZoneId.systemDefault()).toInstant());
    }

    public static Date asDate(LocalDateTime localDateTime) {
        return Date.from(localDateTime.atZone(ZoneId.systemDefault()).toInstant());
    }

    public static LocalDate asLocalDate(Date date) {
        return Instant.ofEpochMilli(date.getTime()).atZone(ZoneId.systemDefault()).toLocalDate();
    }

    public static LocalDateTime asLocalDateTime(Date date) {
        return Instant.ofEpochMilli(date.getTime()).atZone(ZoneId.systemDefault()).toLocalDateTime();
    }

    public static LocalDateTime dayStart(LocalDate date) {
        return LocalDateTime.of(date, LocalTime.MIN);
    }

    public static LocalDateTime dayEnd(LocalDate date) {
        return LocalDateTime.of(date, LocalTime.MAX);
    }

    public static Date dayStart() {
        return asDate(LocalDateTime.of(LocalDate.now(), LocalTime.MIN));
    }

    public static Date dayEnd() {
        return asDate(LocalDateTime.of(LocalDate.now(), LocalTime.MAX));
    }

    public static LocalDateTime monthFirst(LocalDateTime date) {
        return date.with(TemporalAdjusters.firstDayOfMonth());
    }

    public static LocalDateTime monthEnd(LocalDateTime date) {
        return date.with(TemporalAdjusters.lastDayOfMonth());
    }

    public static LocalDate monthFirst(LocalDate date) {
        return date.with(TemporalAdjusters.firstDayOfMonth());
    }

    public static LocalDate monthEnd(LocalDate date) {
        return date.with(TemporalAdjusters.lastDayOfMonth());
    }

    public static Date monthStart() {
        LocalDate localDate = LocalDate.now().with(TemporalAdjusters.firstDayOfMonth());
        return asDate(LocalDateTime.of(localDate, LocalTime.MIN));
    }

    public static Date monthEnd() {
        LocalDate localDate = LocalDate.now().with(TemporalAdjusters.lastDayOfMonth());
        return asDate(LocalDateTime.of(localDate, LocalTime.MAX));
    }

    /**
     * 获取年月日时间
     *
     * @param dat 传入时间 如：2020-01-03 01:02:03
     * @return 处理后的时间 如：2020-01-03 00:00:00
     */
    public static Date getYmd(Date dat) {
        return new DateTime(getDateYMD(dat)).toDate();
    }

    /**
     * 获取中文年月日
     *
     * @param dat 传入时间：如：2020-01-03 01:02:03
     * @return 返回时间：如：2020年01月03日
     */
    public static String getYmdChinese(Date dat) {
        return new DateTime(dat).toString(YMD_CHINESE);
    }

    public static Long betweenYears(LocalDate start, LocalDate end) {
        return ChronoUnit.YEARS.between(start, end);
    }

    public static Long betweenMonth(LocalDate start, LocalDate end) {
        return ChronoUnit.MONTHS.between(start, end);
    }

    public static Long betweenDays(LocalDate start, LocalDate end) {
        return ChronoUnit.DAYS.between(start, end);
    }

    public static Long betweenYears(Date start, Date end) {
        return ChronoUnit.YEARS.between(asLocalDate(start), asLocalDate(end));
    }

    public static Long betweenMonths(Date start, Date end) {
        return ChronoUnit.MONTHS.between(asLocalDate(start), asLocalDate(end));
    }

    public static Long betweenDays(Date start, Date end) {
        return ChronoUnit.DAYS.between(asLocalDate(start), asLocalDate(end));
    }

    public static Long betweenYears(String start, String end) {
        return diffTime(start, end, YMD2_SDF, ChronoUnit.YEARS);
    }

    public static Long betweenMonths(String start, String end) {
        return diffTime(start, end, YMD2_SDF, ChronoUnit.MONTHS);
    }

    public static Long betweenDays(String start, String end) {
        return diffTime(start, end, YMD, ChronoUnit.DAYS);
    }

    public static Long diffTime(String start, String end, String pattern, ChronoUnit unit) {
        java.time.format.DateTimeFormatter formatter = java.time.format.DateTimeFormatter.ofPattern(pattern);
        LocalDate startDate = LocalDate.parse(start, formatter);
        LocalDate endDate = LocalDate.parse(end, formatter);
        return unit.between(startDate, endDate);
    }

    public static LocalDate getLocalDate(Date dat) {
        return dat.toInstant()
                .atZone(ZoneId.systemDefault())
                .toLocalDate();
    }

    /**
     * 根据提供的日期获取当日初始时间
     *
     * @param dat 时间
     * @return 当日初始时间
     */
    public static Date getDayStartTime(Date dat) {
        return Date.from(DateUtil.dayStart(getLocalDate(dat)).atZone(ZoneId.systemDefault()).toInstant());
    }

    /**
     * 根据提供的日期获取当日结束时间
     *
     * @param dat 时间
     * @return 当日结束时间
     */
    public static Date getDayEndTime(Date dat) {
        return Date.from(DateUtil.dayEnd(getLocalDate(dat)).atZone(ZoneId.systemDefault()).toInstant());
    }


    public static Date formatDate(String dat, String pattern) throws ParseException {
        SimpleDateFormat formatter = new SimpleDateFormat(pattern);
        return formatter.parse(dat);
    }

}
